                        //如果时间在误差范围类的就不同步时间，避免时间越级跳位
                        djs = new_djs-1;
                    }
                } else{
                    djs = djs-1;
                }
                if(djs <= 1){
                    var st = $(n).data("style");
                    if(st == "zt"){
                        $(n).html("报价暂停");
                    } else{
                        $(n).html("");//倒计时结束 显示内容设置
                    }
                    return;
                }
                var result = Utils.getTimeCountDownResult(djs);
                $(n).data("value", djs);
                $(n).html($(n).data("title") + "："+result);
            });
        }
        if(!!window["setIntervalId"]) {
            clearInterval(window["setIntervalId"]);
        }
        countDownDjs();
        /*倒计时*/
        window["setIntervalId"] = setInterval(countDownDjs, 1000);
        $(window).unbind("focus");
        $(window).focus(function(){
            Bid.getBidInfo(0);
        });

    });
